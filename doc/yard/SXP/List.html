def ==(other)
  case other
    when List
      self.length == other.length && to_a == other.to_a
    when other.respond_to?(:to_list)
      other.to_list == self
    else
      false
  end
end

def each(&block)

  while pair != nil
    block.call(pair.head)
    pair = pair.tail
  end
  self

def first(count = nil)
  case
    when count.nil?
      @pair.head unless empty?
    when count == 1
      empty? ? [] : [first]
    when count > 1
      empty? ? [] : to_a.first(count)
  end
end

    when Pair
      @pair = other_list

      @pair = other_list.to_pair

      @pair = nil
      unshift(*other_list)
    else
      # TODO
  end
  self
end
