def each(&block)

  while pair != nil
    block.call(pair.head)
    pair = pair.tail
  end
  self
end

def first(count = nil)
  case
    when count.nil?
      @pair.head unless empty?
    when count == 1
      empty? ? [] : [first]
    when count > 1
      empty? ? [] : to_a.first(count)
  end
end
