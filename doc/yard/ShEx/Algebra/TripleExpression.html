def maximum
  @maximum ||= begin
    op = operands.detect {|o| o.is_a?(Array) && o.first == :max} || [:max, 1]
    op[1] == '*' ? Float::INFINITY : op[1]
  end
end

def triple_constraints
  @triple_contraints ||= operands.select do |o|
    o.is_a?(TripleExpression)
  end.
  map(&:triple_constraints).
  flatten.
  uniq
end
