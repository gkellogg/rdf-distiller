def self.registered(app)
  options = {}
  app.set :linkeddata_options, options
  app.use(Rack::LinkedData::ContentNegotiation, options)
  app.helpers(Sinatra::LinkedData::Helpers)
  app.send(:include, ::RDF)
  app.send(:include, ::LinkedData)
end
