def to_rdf_json
  case
    when has_datatype?
      {:type => :literal, :value => value.to_s, :datatype => datatype.to_s}
    when has_language?
      {:type => :literal, :value => value.to_s, :lang => language.to_s}
    else
      {:type => :literal, :value => value.to_s}
  end
end
