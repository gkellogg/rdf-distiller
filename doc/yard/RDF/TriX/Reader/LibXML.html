def initialize_xml(options = {})
  require 'libxml' unless defined?(::LibXML)
  @xml = case @input
    when File   then ::LibXML::XML::Document.file(@input.path)
    when IO     then ::LibXML::XML::Document.io(@input)
    else ::LibXML::XML::Document.string(@input.to_s)
  end
end
