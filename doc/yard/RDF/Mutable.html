def entail!(*rules, &block)
  rules = %w(subClassOf subPropertyOf domain range).map(&:to_sym) if rules.empty?
  statements = []

  self.each do |statement|
    rules.each do |rule|
      statement.entail(rule) do |st|
        statements << st
      end
    end
  end
  self.insert(*statements)
  self
end
