def self.detect(sample)
  !!sample.match(%r(
    (?:\s*(?:<[^>]*>) | (?:_:\w+))                          # Subject
    \s*
    (?:\s*<[^>]*>)                                          # Predicate
    \s*
    (?:(?:<[^>]*>) | (?:_:\w+) | (?:"[^"\n]*"(?:^^|@\S+)?)) # Object
    \s*
    (?:\s*(?:<[^>]*>) | (?:_:\w+))                          # Context
    \s*\.
  )x) && !(
    sample.match(%r(@(base|prefix|keywords)|\{)) ||         # Not Turtle/N3/TriG
    sample.match(%r(<(html|rdf))i)                          # Not HTML or XML
  )
end
