def apply(left, right)
  case
    when left.is_a?(RDF::Literal::Numeric) && right.is_a?(RDF::Literal::Numeric)
      left * right
    else raise TypeError, "expected two RDF::Literal::Numeric operands, but got #{left.inspect} and #{right.inspect}"
  end
end
