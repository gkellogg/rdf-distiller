def execute(queryable, options = {})
  debug(options) {"Update"}
  raise IOError, "queryable is not mutable" unless queryable.mutable?
  operands.each do |op|
    op.execute(queryable, options.merge(depth: options[:depth].to_i + 1))
  end
  queryable
end
