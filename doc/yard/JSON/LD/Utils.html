def as_resource(id, base = nil)
  @nodes ||= {} # Re-use BNodes
  if id[0,2] == '_:'
    (@nodes[id] ||= RDF::Node.new(id[2..-1]))
  elsif base
    base.join(id)
  else
    RDF::URI(id)
  end
end

def blank_node?(value)
  case value
  when nil    then true
  when String then value[0,2] == '_:'
  else
    (node?(value) || node_reference?(value)) && value.fetch('@id', '_:')[0,2] == '_:'
  end
end

def node?(value)
  value.is_a?(Hash) &&
    (value.keys & %w(@value @list @set)).empty? &&
    !(value.keys - ['@id']).empty?
end
